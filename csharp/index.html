🔹 Katalogi (Directory i DirectoryInfo)

using System.IO;

string sciezka = Path.Combine(Environment.GetFolderPath(Environment.SpecialFolder.Desktop), "WdP_Zadanie12.1");

if (!Directory.Exists(sciezka))
{
    Directory.CreateDirectory(sciezka);
    Console.WriteLine("Utworzono katalog");
}
else
{
    Console.WriteLine("Katalog już istnieje");
}

DirectoryInfo info = new DirectoryInfo(sciezka);
if (info.Exists)
{
    Console.WriteLine($"Data utworzenia: {info.CreationTime}");
    Console.WriteLine($"Ostatni dostęp: {info.LastAccessTime}");
    Console.WriteLine($"Dysk: {Path.GetPathRoot(info.FullName)}");
    Console.WriteLine($"Macierzysty: {info.Parent}");
    Console.WriteLine($"Pełna ścieżka: {info.FullName}");
}

🔹 Przydatne metody do zadań z kolokwium

// 📁 Pliki - File
if (File.Exists("Nowy.txt"))
{
    string[] linie = File.ReadAllLines("Nowy.txt");  // odczyt do tablicy
    foreach (string linia in linie)
        Console.WriteLine(linia);

    File.Copy("Nowy.txt", "Nowy_kopia.txt", true);    // kopia
    File.Delete("Nowy_kopia.txt");                    // usunięcie
}
else
{
    Console.WriteLine("Plik nie istnieje!");
}

// 📂 Katalog - Directory
string katalog = Path.Combine(
    Environment.GetFolderPath(Environment.SpecialFolder.Desktop),
    "WdP_Zadanie12.1");

if (!Directory.Exists(katalog))
{
    Directory.CreateDirectory(katalog);
    Console.WriteLine("Utworzono katalog.");
}
else
{
    Console.WriteLine("Katalog już istnieje.");
}

// 📊 Informacje o katalogu - DirectoryInfo
DirectoryInfo dir = new DirectoryInfo(katalog);
if (dir.Exists)
{
    Console.WriteLine($"Data utworzenia: {dir.CreationTime}");
    Console.WriteLine($"Ostatni dostęp: {dir.LastAccessTime}");
    Console.WriteLine($"Dysk: {Path.GetPathRoot(dir.FullName)}");
    Console.WriteLine($"Katalog macierzysty: {dir.Parent}");
    Console.WriteLine($"Pełna ścieżka: {dir.FullName}");
}
